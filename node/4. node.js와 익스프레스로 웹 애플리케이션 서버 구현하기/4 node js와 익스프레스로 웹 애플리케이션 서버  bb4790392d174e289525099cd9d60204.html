3.4 동적으로 응답하기

앞 작성한 서버는 브라우저에 호스트 접속하면 언제나 같은결과를 보여줍니다. user() 함수를 수정해서 매겨변수에 따라 동적으로 응답이 변경되도록 함

user()함수는 응답으로 고정된 name와 age정보를 반환함

url의 query 부분에 name와 age 정보를 추가함

const http = require("http");
const url = require("url");
http
  .createServer((req, res) => {
    const path = url.parse(req.url, true).pathname;
    res.setHeader("Content-Type", "text/html");
    if (path === "/user") {
      user(req, res); // 1 
    } else if (path === "/feed") {
      feed(req, res); // 2
    } else {
      notFound(req, res); // 3
    }
  })
  .listen("3000", () => console.log("라우터를 만들어보자!"));

//////////////////////////////////////

const user = (req, res) => {
  const userInfo = url.parse(req.url, true).query;
  // 1 쿼리스트링데이터를 userInfo에 할당
  res.end(`[user] name : ${userInfo.name}, age: ${userInfo.age}`);
  // 2 결과값으로 이름과 나이 설정
};

//////////////////////////////////////

const feed = (req, res) => {
  res.end(`<ul>
        <li>picture1</li>
        <li>picture2</li>
        <li>picture3</li>
      </ul>
      `);
};

const notFound = (req, res) => {
  res.statusCode = 404;
  res.end("404 page not found");
};

http://localhost:3000/user?name=lhg &age=20

쿼리스트링으로 이용하여 값을 넣음
