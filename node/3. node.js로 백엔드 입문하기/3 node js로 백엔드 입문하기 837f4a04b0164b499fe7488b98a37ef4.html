3.3 createServer()리팩터링하기

리팩터링 : 동작결과를 변경하지않으면서 코드의 구조를 재조정하는 작업(가독성, 유지보수 업)

현재는 createServer()안의 콜백함수에 모든 코드를 다 추가해야 하므로 좋지않음

라우팅 이후 처리를 별도의 함수를 만들어 처리하는 코드로 리팩토링

const http = require("http");
const url = require("url");
http
  .createServer((req, res) => {
    const path = url.parse(req.url, true).pathname;
    res.setHeader("Content-Type", "text/html");
    if (path === "/user") {
      user(req, res); // 1 user() 함수 실행
    } else if (path === "/feed") {
      feed(req, res); // 2 feed() 함수 실행
    } else {
      notFound(req, res); // 3 notFound()함수 실행
    }
  })
  .listen("3000", () => console.log("라우터를 만들어보자!"));

const user = (req, res) => {
  res.end(`[user] name : andy, age: 30`);
};

  res.end(`<ul>
        <li>picture1</li>
        <li>picture2</li>
        <li>picture3</li>
      </ul>
      `);
};

const notFound = (req, res) => {
  res.statusCode = 404;
  res.end("404 page not found");
};

  1. path가 없을때 notFound()함수로 분리
